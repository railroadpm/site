{{- $calculated_base_key := printf "%s%s%s" $.m_key "_" $.ap_key -}} {{- $total_key := printf "%s%s" $calculated_base_key "_Total" -}} {{- $count_key := printf "%s%s" $calculated_base_key "_Count" -}} {{- $avg_key := printf "%s%s" $calculated_base_key "_Avg" -}} {{- $week_key_for_avg := cond (eq $.ap_index 0) "Qtr" "Month" -}} {{- $weeks_all := $.data.Dimensions.Period.Week -}} {{- $weeks_publishable := sort (where $weeks_all "DoNotPublish" "ne" true) "Key" "asc" -}} {{- $weeks_to_avg := where $weeks_all $week_key_for_avg "eq" $.ap_key -}} {{- $.scratch.Set $total_key 0 -}} {{- $.scratch.Set $count_key 0 -}} {{- $.scratch.Set $avg_key nil -}} {{- range $week := $weeks_to_avg -}} {{- with (index (index $.rpts $week.Key) $.m_key) -}} {{- $.scratch.Add $total_key . -}} {{- $.scratch.Add $count_key 1 -}} {{- end -}} {{- end -}} {{- $numerator := $.scratch.Get $total_key -}} {{- $denominator := $.scratch.Get $count_key -}} {{- with $denominator -}}{{- $.scratch.Set $avg_key ((div (float $numerator) (float .)) | lang.NumFmt $.m_dp) -}}{{- end -}}
